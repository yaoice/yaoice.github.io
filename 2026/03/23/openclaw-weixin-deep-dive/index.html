4.1 Runtime жіЁе…ҘжЁЎејҸ

setWeixinRuntime() е°ҶжЎҶжһ¶жҸҗдҫӣзҡ„ PluginRuntime еӯҳе…ҘжЁЎеқ—зә§е…ЁеұҖеҸҳйҮҸгҖӮиҝҷдҪҝеҫ—еҗҺз»ӯжүҖжңүжЁЎеқ—йғҪиғҪйҖҡиҝҮ getWeixinRuntime() и®ҝй—®жЎҶжһ¶иғҪеҠӣпјҲи·Ҝз”ұгҖҒдјҡиҜқгҖҒеӘ’дҪ“еӯҳеӮЁзӯүпјүпјҢиҖҢдёҚйңҖиҰҒеңЁеҮҪж•°и°ғз”Ёй“ҫдёӯеұӮеұӮдј йҖ’пјҡ

// runtime.ts вҖ” ејӮжӯҘзӯүеҫ…жЁЎејҸ
export async function waitForWeixinRuntime(timeoutMs = 10_000): Promise<PluginRuntime> {
  const start = Date.now();
  while (!pluginRuntime) {
    if (Date.now() - start > timeoutMs) throw new Error("timeout");
    await new Promise(r => setTimeout(r, 100));  // 100ms иҪ®иҜўй—ҙйҡ”
  }
  return pluginRuntime;
}

9.1 AES-128-ECB еҠ и§ЈеҜҶ

// еҠ еҜҶ
export function encryptAesEcb(plaintext: Buffer, key: Buffer): Buffer {
  const cipher = createCipheriv("aes-128-ecb", key, null);
  return Buffer.concat([cipher.update(plaintext), cipher.final()]);
}

// и§ЈеҜҶ
export function decryptAesEcb(ciphertext: Buffer, key: Buffer): Buffer {
  const decipher = createDecipheriv("aes-128-ecb", key, null);
  return Buffer.concat([decipher.update(ciphertext), decipher.final()]);
}

// йў„жөӢеҜҶж–ҮеӨ§е°ҸпјҲPKCS7 еЎ«е……пјү
export function aesEcbPaddedSize(plaintextSize: number): number {
  return Math.ceil((plaintextSize + 1) / 16) * 16;
}
